'use strict';
 
var listenForTokenUpdates = function (context) {
 
    // listen for token updates
    context._generator.on('token', function(token){
        console.log('New token for %s: %s',
            token.user,
            token.accessToken
        );
    });
 
};
 
var generateEmail = function (context, fileName, sensorValue){
 
	return {
	    'from': context._config.mailSender.from,
	    'to': context._config.mailSender.to,
	    'subject': context._config.mailSender.subject,
	    'attachments': [
		    {
			    'filename': fileName,
			    'content': context._fs.createReadStream(fileName)
		    }
	    ],
	    'html': '<b>Light index: ' + sensorValue + ' âœ”</b>'
    };
 
};
 
function Email(config, nodemailer, xOAuth2, fs){
	
	this._config 		= config;
	this._nodemailer 	= nodemailer;
	this._xOAuth2 		= xOAuth2;
	this._mailOptions 	= {};
	this._fs = fs;
	this.isAllowedToSendMoreEmails = true;
	
	this._generator		= this._xOAuth2.createXOAuth2Generator(
							this._config.mailCredentials
							);
 
	var transporterOptions = {	
								service: this._config.mailService,
						    	auth: {
						        	xoauth2: this._generator
						    	}
							};
 
	this._transporter 	= this._nodemailer.createTransport(transporterOptions);
 
	listenForTokenUpdates(this);
 
}
 
Email.prototype.send = function(fileName, sensorValue){
 
	if (this.isAllowedToSendMoreEmails !== true) {
 
		console.log('Waiting to send more emails...');
 
	} else {
 
		var _this = this;
 
		this._mailOptions = generateEmail(this, fileName, sensorValue);
		this._transporter.sendMail(this._mailOptions, function (error) {
 
			if (error) {
				console.log(error);
			} else {
				console.log('===Email Sent...!');
				_this.isAllowedToSendMoreEmails = false;
				setTimeout(function () {
 
					_this.isAllowedToSendMoreEmails = true;
					console.log('===Allowed to send more emails');
 
				}, _this._config.timeBetweenEmails);
			}
 
		});
 
	}
 
};
 
module.exports = Email;
